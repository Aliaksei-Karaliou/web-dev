function makeUsable (url, replaces) {
  var parts = url.split('/');
  var i;

  for (i = 0; i < replaces.length; i += 2) {
    parts[4] = parts[4].replace(replaces[i], replaces[i + 1]);
  }

  return parts.join('/');
}
